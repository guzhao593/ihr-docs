# 开始使用

# 写在前面

配置平台是基于Sahara、Sahara-studio实现的 可配置页面工具 ，我们秉承 Ant 的设计价值观，致力于在设计规范和基础组件的基础上，继续向上构建，提炼出典型模板/业务组件/配套设计资源，进一步提升企业级中后台产品设计研发过程中的『用户』和『设计者』的体验。随着『设计者』的不断反馈，我们将持续迭代，逐步沉淀和总结出更多设计模式和相应的代码实现，阐述中后台产品模板/组件/业务场景的最佳实践，也十分期待你的参与和共建。

# 谁在使用

目前美云智数的 iHR 和 财务云 的开发者正在使用 Sahara 的研发模式，可以联系我们

# 初始化 CPF

由于配置平台暂未实现独立部署，而是采用组件引用的方式，故例如发送请求等操作需要产品/项目注入支持

import '@ipeople/sahara/dist/styles/sahara.css';                  // 引入sahara样式
import '@ipeople/sahara-studio/dist/styles/sahara-studio.css';    // 引入sahara-studio样式

import views from "@/components/common/i-lookup-views";           // sahara通用选择器配置信息
import { remotes } from "@/config/sahara/selector";               // sahara下拉选择器请求信息

installCpf(Vue, {
  url: {
    // CPF以来的接口，需要配置代理地址
    originURL: "http://10.18.61.34", // 原始请求
    baseURL: "http://10.18.61.34:8080/base", // 基础数据请求，例如字典请求
    cpfURL: "http://10.18.61.34:8082/expense", // 配置平台请求
    expenseUrl: "http://10.18.61.34:8082/expense",
    backendURL: "http://10.18.61.34:8080/backend"
  },
  components: {
    lookup: views,                                                // 用于设计时通用选择器配置关联业务
    remotes                                                       // 用于设计时下拉选择器配置数据来源为远程时的关联业务
  },
  designSettings: {},                                            // 注册进设计器的额外配置，包括但不限于设计器需引用的自定义函数、台账设计器中的路由路径等
  runtimeSettings: {},                                             // 注册进运行时的额外配置
  // 为了兼容财务云多窗口模式，提供跳转新页面的方法
  $openTab: (routeUrl, title) => {
    // 财务云可不传
    window.location.href = routeUrl;
  },
  $closeTab: () => {
    // 关闭配置平台菜单的方式
    window.history.go(-1);
  },
  headerOptions(){
    return {
      // 项目自定义请求头，主要用于token传入，示例如下
      'x-auth-token': localStorage.getItem('token')
    }
  },
  // 获取当前员工信息
  getUserID() {
    const userInfo = localStorage.getItem("userInfo");
    return JSON.parse(userInfo).id;
  }
});

# 注册 cpf 页面路由

为避免路由冲突，产品/项目上自定义运行时尽量不要以cpf做路径

// 注意避免路由冲突
// router/index.js
import { cpfRoutes } from "@ipeople/sahara-cpf-test/dist_pkg/sahara-cpf";

Vue.use(Router);

export default new Router({
  routes: [
    cpfRoutes // 配置平台设计器所有菜单路由，可自由配置引用
    // ...other
  ]
});

# 在产品/项目初始Sahara时注册cpf组件

// 配置平台内置selector的配置信息需要注册进产品
import { cpfSaharaConfig } from '@ipeople/sahara-cpf-test/dist_pkg/sahara-cpf'
import { selectorConfig } from './selector' // 产品自己的selector配置

Vue.use(Sahara, {
  // 其余配置
  selector: {
    remotes: { ...selectorConfig.remotes, ...cpfSaharaConfig.selector.remotes }
  }
})
