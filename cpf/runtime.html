# 说明

在引入配置平台之后项目上可以通过直接配置配置平台提供的菜单路径访问设计器进行配置页面，在此基础上产品/项目团队需独立提供一套运行时文件用来承载设计数据并进行展示。

目前配置平台提供了台账/单据/表单三种技术模板，推荐项目上至少为每种技术模板编写一套运行时文件。

# 注意

由于目前配置平台暂未实现独立部署，故运行时所依赖的接口需项目上提供，具体清单如下

接口名 接口路径 说明
getValidation /validation/{validationId} 获取当前校验规则
getBusinessView /business/views/{viewId} 获取当前配置视图的数据

# 配置平台提供的可被使用的运行时方法

# Events

事件名称 说明 回调参数
this.submit 触发当前配置的视图数据的校验并获取数据 -

# Methods

方法名 说明 参数
this.$forms.setModel 设置当前配置视图的数据 (modelData: Object)
this.$forms.getModel 获取当前配置视图的数据 -
this.$forms.setReadonly 设置当前视图的只读状态 (isReadonly: Boolean)
this.$forms.validate() 执行当前视图的校验 -
this.$forms.clearValidate() 清除当前视图的校验状态 -

# 推荐运行时目录结构

.
├── view-runtime (存放所有运行时相关的文件)
│   ├── form.vue (表单运行时文件)
│   ├── list.vue (列表运行时台账文件)
│   ├── modal.vue (列表运行时表单文件)
│   ├── order.vue (单据运行时文件)
│   ├── config.js (存放运行时依赖的配置文件)
│   ├── components (存放运行时依赖的定制组件)
└── api.js

# 单据运行时示例

<script>
// 引入试运行文件所依赖的接口
import req from '@/fssc/cpf/api';
// 需要从Sahara-studio引入render函数
import { Renderer } from '@ipeople/sahara-studio';

// 本示例表单模板/单据模板的试运行在一个文件实现，故有两个接口（需要在项目/产品注册）
const DETAILAPI_MAP = {
  order: 'getOrderDetail', // 单据模板
  form: 'getBusinessFormDetail', // 表单模板
};

/**
 * 抽离出获取运行时视图数据的函数，接口需要在项目/产品注册
 * code：业务功能编码
 * viewType：视图类型
*/
const fetchView = async (code, viewType) => {
  const { data } = await req('getBusinessView', { code, viewType }) || {};
  const { regulations, ...viewNode } = JSON.parse(data.configContent || '{}').viewModel || {};
  return {
    regulations,
    viewNode,
  };
};

let time = null;

export default {
  mixins: [Renderer],

  props: {
    from: {
      type: String,
      default: 'order',
    },
  },

  render(h) {
    // 配置平台配置出来的视图渲染render,针对需要对配置平台的表单进行扩展在这个函数实现
    return (
      <layout class="runtime-layout">
        <div>{Renderer.render.call(this, h)}</div>
      </layout>
    );
  },

  data() {
    return {
      code: null, // 当前业务功能编码
      orderId: null, // 当前单据/表单的主键id，项目上自行定义
      dataModelId: null, // 当前数据模型id
      validationId: null, // 当前校验模型id
      layoutLoading: null, // loading进度条
      model: null,
    };
  },
  async mounted() {
    // 渲染业务功能视图/校验，为标准代码，可以直接使用
    this.code = 'PA_LOAN' // 业务功能编码
    const viewType = 'edit'; // 以渲染填单页为例
    const { regulations, viewNode } = await fetchView(this.code, viewType);
    const { data: validationData } = this.validationId
      ? await req('getValidation', { validationId: this.validationId })
      : { data: {} };

    time = Date.now();

    this.launch({
      viewNode,
      regulations,
      validations: (validationData.validationNodes || []).map(validation => {
        return {
          ...validation,
          conditions: JSON.parse(validation.conditions),
        };
      }),
    });

    // 针对项目上需要填充表单数据的行为，示例如下
    if (this.orderId) {
      // 如果存在单据id,则获取视图数据填充进表单
      const initModelData = await this.getFormDetail(this.orderId, viewType) || {};
      this.$forms.setModel(initModelData);
      setTimeout(() => this.$forms.clearValidate(), 0);
    }
  },

  // $forms 在渲染完成之后才能访问
  rendered() {
    console.log(`Renderer used ${Date.now() - time}ms`);
    this.model = this.$forms.getModel(); // 通过此方法调用单据的数据对象
  },

  methods: {
    // 单据保存-存储引擎
    async postFormStorage(cb, valid = true) {
      const formData = await this.submit({ valid }); // valid判断是否需要前端校验
      // 产品上实现的统一单据保存接口
      return req('postExpStorage', {
        code: this.code,
        id: this.orderId || 'null',
        ...formData,
      }).then(res => {
        this.orderId = res.data.result;
        console.log('单据orderId', this.orderId);
        cb && cb();
      }).catch(err => {

        this.layoutLoading && this.layoutLoading.close();
      }).finally(() => {
      });
    },
    // 根据orderId获取表单数据
    getFormDetail(id, viewType = 'edit') {
      return req('getFormInstanceViewData', {
        functionCode: this.code,
        pkId: id,
        viewType,
      }).then(res => {
        this.formDetail = res.data || {};
        return res.data;
      }).catch(err => {
        // this.$message.error(err);
      });
    },
  },
};
</script>

# 列表运行时示例

# List文件

<script>
// cpf请求体
import req from '@/fssc/cpf/api';
import { Renderer } from '@ipeople/sahara-studio';
import { settings } from '../view-runtime/config';
// 列表表单运行时
import Modal from './modal';
// 配置平台提供了导入配置，但导入组件依赖于产品/项目，故需引入产品/项目的的公用导入组件
import CommonImport from '../common-import';

const DEFAULT_SOURCETYPE = 'dm'; // 默认为列表类型数据源为单据列表，适应已配的列表运行时菜单页面

// 跳转公共导入配置的方法
export const commonImport = function (businessCode, templateName) {
  this.$openTab(`#/cpf/import/common-import?businessCode=${businessCode}&templateName=${templateName}`, '导入');
};

// 初始化列表视图
export const initListPageViewNode = function (viewNode, {
  search,
  functionCode,
  tableHandler,
  formSearchHandler,
}) {
  this.targetTable = {};
  const refs = [];
  if (!viewNode.content) {
    this.$message.error('暂无视图');
    return;
  }

  // 标记搜索区和表格区，绑定请求函数
  viewNode.content.forEach(block => {
    const { sourceNode, ref } = block.component.options;
    refs.push(ref);

    if (block.component.name === 'i-form-search') {
      // 查询按钮的 search 绑定在 render.js 中执行
      formSearchHandler && formSearchHandler(block.component.options.sourceNode);
    } else {
      sourceNode.children.forEach(formItem => {
        const [controller = {}] = formItem.children;
        if (controller.tag === 'i-table') {
          this.targetTable = { ref, prop: formItem.attrs.prop };
          controller.props.editing = true;

          const editor = Object.entries(controller.props.editor).reduce((obj, [key, value]) => {
            if (value.type === 'button') {
              obj[key] = value;
            } else if (value.type === 'lookup') {
              const props = value.props = value.props || {};
              props.readonly = true;
              obj[key] = value;
            }
            return obj;
          }, {});
          controller.props.editor = editor;

          controller.props.total = 0;
          controller.props.info = {};
          // 增加导出配置
          controller.props.exportExcel && (controller.props.exportExcel = {
            config: {
              fileName: '',
            },
            reqConfig: {
              taskUrl: this.originUrl,
              url: `${this.originUrl}/common/exports/list/${functionCode}/list-data?pageSize={pageSize}&pageIndex={pageIndex}`,
              method: 'post',
            },
          });
          this.tableProps = controller.props;
          controller.on = controller.on || {};
          controller.on['page-change'] = search;
          controller.ref = 'table';
          tableHandler && tableHandler(controller);
        }
      });
    }
  });

  viewNode.struct = [
    {
      name: 'group',
      type: 'group',
      children: refs.map(name => ({ name })),
    },
  ];
};

// 初始化查询字段
export const initQueryOptions = function ({ rangeList, defaultValueList }) {
  this.rangeListMap = rangeList.reduce((acc, cur) => ({
    ...acc,
    [cur.prop.split('.').join('$')]: cur.range,
  }), {});
  this.defaultQuery = defaultValueList.reduce((acc, cur) => ({
    ...acc,
    [cur.prop.split('.').join('$')]: cur.value,
  }), {});
};

// 运行时查询条件数据格式化
export const formatQuery = function (query) {
  const { rangeListMap } = this;
  const isUndef = val => val == null || val === '';

  return Object.keys(query).reduce((acc, key) => {
    let value = query[key];
    if (isUndef(value)) {
      return acc; // 移除空值字段
    }
    if (rangeListMap[key]) {
      // 查询字段为范围查询时需要特殊处理拼装成数组
      const range = rangeListMap[key];
      const hasStart = range.indexOf('start') > -1;
      const hasEnd = range.indexOf('end') > -1;

      let [startValue, endValue] = Array.isArray(value) ? value : [value, value];
      startValue = hasStart ? startValue : '$';
      endValue = hasEnd ? endValue : '$';

      if (
        startValue === '$' && endValue === '$' ||
        (isUndef(startValue) && isUndef(endValue))
      ) {
        // 若始末数据都为无穷($)，或始末数据都不存在，则忽略该请求条件
        return acc;
      }
      value = JSON.stringify([startValue, endValue]);
    }
    return { ...acc, [key]: value };
  }, {});
};

export default {
  mixins: [Renderer],

  data() {
    return {
      settings,
      query: {},
      defaultQuery: {},
      targetTable: {},
      tableProps: {},
      pkName: '',
      modalViewModel: null,
      loading: false,
      isPopup: false,
      rangeListMap: {},
      listTplDataSourceType: '',
      templateName: '',
    };
  },

  computed: {
    userInfo() {
      return this.$store.state.userInfo;
    },

    originUrl() {
      return this.$store.state.baseConfig.cpfURL;
    },

    functionCode() { // 此情况作为组件嵌入到页面中
      let route = this.$attrs.route;
      return route ? route.split('?')[0].split('/').pop() : '';
    },
  },

  watch: {
    '$route.params': {
      handler(to, from) {
        this.init();
      },
      deep: true,
    },
  },

  mounted() {
    this.init();
  },

  rendered() {
    this.search();
  },

  methods: {
    async init() {
      const { code: functionCode } = this.$route.params;
      const { sourceType: listTplDataSourceType } = this.$route.query;
      const { data: { form, list, pkName } } = await req('getListEngineEntry', { functionCode: functionCode || this.functionCode }) || {};
      const {
        viewModel: { regulations, ...viewNode } = {},
        templateConfig: { rangeList = [], defaultValueList = [], formEnterType } = {},
      } = JSON.parse(list.configContent || '{}');

      this.templateName = list.viewName;
      this.isPopup = formEnterType === 'popup';
      this.pkName = pkName;
      this.listTplDataSourceType = listTplDataSourceType || DEFAULT_SOURCETYPE;
      initQueryOptions.call(this, { rangeList, defaultValueList });
      initListPageViewNode.call(this, viewNode, { search: this.search, functionCode });

      await this.launch({
        viewNode,
        regulations,
        model: this.value,
      });
      if (this.isPopup) {
        // 若是弹窗表单，则在外层解析表单视图
        this.$nextTick(() => {
          const { regulations, ...viewNode } = JSON.parse(form.configContent || '{}').viewModel || {};
          this.modalViewModel = Object.freeze({ regulations, viewNode });
        });
      }
    },

    // 处理重置
    reset() {
    },

    async search(query) {
      let { code: functionCode } = this.$route.params;
      functionCode = functionCode || this.functionCode;
      const { ref, prop } = this.targetTable;
      if (!ref || !prop) {
        throw new Error(`缺失表格信息: ref: ${ref}, prop: ${prop}`);
      }

      this.loading && this.loading.close();
      this.loading = this.$loading({
        target: '.i-table',
        text: '加载中',
      });
      this.query = Object.assign({}, this.query, query, this.defaultQuery);

      try {
        const { data: { data, total, info } } = await req('postListEngineData', {
          pageIndex: 1,
          pageSize: 10,
          ...formatQuery.call(this, this.query),
          functionCode,
        });
        // 同步表格信息
        this.$forms.setModel(ref, {
          [prop]: Object.freeze(data.map(key => {
            this.tableProps.columns.forEach(col => {
              if (!col.display && Array.isArray(key[col.prop])) {
                key[col.prop] = key[col.prop].join(',');
              }
            });
            return key;
          })),
        });
        this.tableProps.info = info;
        this.tableProps.total = total;
      } finally {
        this.loading && this.loading.close();
      }
    },
    // 设计器按钮方法为路由跳转时的函数，产品/项目上可重写
    routerPush(vm, opts) {
      let path;
      const name = vm.$options.name;
      if (opts.path.includes('F.')) {
        path = `#/cpf/runtime/form-runtime/${opts.path.replace('F.', '')}`;
      } else {
        path = `#/cpf/runtime/order-runtime-${opts.type}/${opts.path}`;
      }

      if (name === 'i-table' || name === 'i-toolbar') {
        const title = '新增';
        this.$openTab(path, title);
      } else if (vm.cell) {
        const title = opts.type === 'edit' ? '编辑' : '详情';
        const { row } = vm.cell;
        const pkId = row[this.pkName];
        this.$openTab(`${path}?orderId=${pkId}`, title);
      }
    },
    // 列表配置新增默认函数，产品/项目上可重写
    createRow(vm, ev) {
      let { code: functionCode } = this.$route.params;
      functionCode = functionCode || this.functionCode;
      // 跳转路由时默认新增
      if (!this.isPopup) {
        this.$openTab(`#/cpf/runtime/list-form/${functionCode}?pageType=create&sourceType=${this.listTplDataSourceType}`, '新增');
        return;
      }
      // 弹出框时默认新增
      const title = '新增';
      const opts = {
        title,
        component: Modal,
        addition: {
          ...this.modalViewModel,
          pageType: 'create',
          functionCode,
          sourceType: this.listTplDataSourceType,
        },
        beforeWriteback({ data: [data] }, cb) {
          // 配置平台新增接口
          req('postListEngineFormData', {
            ...data,
            functionCode,
          }).then(cb).catch(() => cb(false));
        },
      };
      this.$lookup(opts).then(() => {
        this.$message.success(`${title}成功`);
        this.search({ pageIndex: 1 });
      });
    },
    // 列表配置编辑默认函数，产品/项目上可重写
    updateRow(vm, ev) {
      let { code: functionCode } = this.$route.params;
      functionCode = functionCode || this.functionCode;
      const { row } = vm.cell;
      const pkId = row[this.pkName];
      if (!this.isPopup) {
        this.$openTab(`#/cpf/runtime/list-form/${functionCode}?pageType=update&id=${pkId}&sourceType=${this.listTplDataSourceType}`, '编辑');
        return;
      }
      const title = '编辑';
      const opts = {
        title,
        component: Modal,
        addition: {
          ...this.modalViewModel,
          pageType: 'update',
          functionCode,
          pkId,
          sourceType: this.listTplDataSourceType,
        },
        beforeWriteback({ data: [data] }, cb) {
          req('putListEngineFormData', {
            ...data,
            functionCode,
            pkId,
          }).then(cb).catch(() => cb(false));
        },
      };
      this.$lookup(opts).then(() => {
        this.$message.success(`${title}成功`);
        this.search({ pageIndex: 1 });
      });
    },
  // 列表配置查看默认函数，产品/项目上可重写
    detailRow(vm, ev) {
      let { code: functionCode } = this.$route.params;
      functionCode = functionCode || this.functionCode;
      const { row } = vm.cell;
      const pkId = row[this.pkName];
      if (!this.isPopup) {
        this.$openTab(`#/cpf/runtime/list-form/${functionCode}?pageType=detail&id=${pkId}&sourceType=${this.listTplDataSourceType}`, '详情');
        return;
      }
      const title = '查看';
      const opts = {
        title,
        component: Modal,
        addition: {
          ...this.modalViewModel,
          pageType: 'detail',
          functionCode,
          pkId,
          sourceType: this.listTplDataSourceType,
        },
        dialogProps: {
          defaultToolbar: false,
        },
      };
      this.$lookup(opts);
    },
  // 列表配置删除默认函数，产品/项目上可重写
    deleteRow(vm, ev) {
      let { code: functionCode } = this.$route.params;
      functionCode = functionCode || this.functionCode;
      const { row } = vm.cell;
      const pkId = row[this.pkName];
      const title = '删除';
      this.loading = this.$loading({
        text: '删除中',
        target: '.i-table',
      });
      req('deleteListEngineFormData', {
        functionCode,
        pkId,
      }).then(() => {
        setTimeout(() => {
          this.loading && this.loading.close();
          this.$message.success(`${title}成功`);
          this.search({ pageIndex: 1 });
        }, 700);
      }).catch(() => {
        this.loading && this.loading.close();
      });
    },

    // 运行时调用导入，项目上可重写
    importRows(vm, ev, businessCode) {
      commonImport.call(this, businessCode, this.templateName);
    },

    // 批量更新函数
    updateRows(vm, ev, opts = {}) {
      const { code: functionCode } = this.$route.params;
      const title = '批量修改';
      const rows = vm.getSelection();

      if (Object.keys(opts).length === 0 || rows.length === 0) return;

      const data = rows.reduce((obj, row) => {
        const pkId = row[this.pkName];
        obj[pkId] = opts;
        return obj;
      }, {});

      req('batchUpdateListEngineFormData', {
        functionCode,
        ...data,
      }).then(() => {
        this.$message.success(`${title}成功`);
        this.search({ pageIndex: 1 });
      });
    },
  },
};
</script>

import req4base from '@/fssc/base/api';
import { getDict } from '@/config/utils';
import { moneyformatter } from '@/js/util';
import { renderAddRowButton } from '@ipeople/sahara-studio';

const getDicts = () => req4base('dictPageQuery', {
  pageIndex: 1,
  pageSize: 1000,
}).then(({ records }) => {
  return records.map(el => ({
    value: el.code,
    label: `${el.name} ${el.code}`,
  }));
});

export function renderPureAddRowButton(h, { handleClick, text, style }) {
  return (
    <div
      style={{
        borderStyle: 'dashed',
        padding: '8px 0',
        width: '100%',
        ...style,
      }}
      class="el-button el-button--default is-plain"
      onClick={handleClick}
    >
      <i class="el-icon el-icon-plus" style="font-size: 12px"></i>
      <span style={{ fontWeight: ' normal', fontSize: '12px' }}>
        {text || '添加'}
      </span>
    </div>
  );
}

export function getSummarys({ columns, data, summaryMap, formatter }) {
  const sums = [];
  columns.forEach((column, index) => {
    const prop = column.property;
    if (index === 0) {
      sums[index] = '合计';
    } else if (summaryMap[prop]) {
      const { type, param } = summaryMap[prop] === true ? { type: 'number' } : summaryMap[prop];
      const value = data.reduce((acc, cur) => Number(acc) + (Number(cur[prop]) || 0), 0);
      if (type === 'rate' || type === 'number') {
        sums[index] = formatter[type] ? formatter[type](param, value) : value;
      }
    }
  });
  return sums;
}

function renderDescription(h, data) {
  const model = this.$forms.getModel();
  return (
    <div className="handler-info">
      {data.map(cur => {
        const text = cur.fields.map(f => model[`${f[0]}`][`${f[1]}`]).join(' - ');
        return (
          <div>{cur.label}：{text}</div>
        );
      })}
    </div>
  );
}

export function renderSummarization(h, data) {
  const name = 'i-form-renderer-summarization';
  data = data || this.getSummarization();
  if (!data.length) return;
  return (
    <div class={name}>
      {data.map(cur => {
        const value = cur.value == null ? 0 : cur.value;
        return (
          <div class={`${name}-item`}>
            <span class={`${name}-item-label`}>{cur.label}：</span>
            <span class={`${name}-item-value`}>{moneyformatter(value)}</span>
          </div>
        );
      })}
    </div>
  );
}

export const settings = {
  'sahara.table.renderFooter.showAddButton': renderAddRowButton, // 注册表格新增按钮
  'sahara.selector.fetchDictOptions': getDict, // 注册选择器获取字典项的方法
  'sahara.selector.fetchDicts': getDicts, //注册选择器获取所有字典的方法
  'sahara.table.summaryMethod': getSummarys, // 注册表格合计的方法
  'sahara.form.item.renderDescription': renderDescription, // 注册form表单的描述信息
  ...(outside || {}).runtimeSettings,
};

# Modal文件

<script>
import req from '@/fssc/cpf/api';
import { Renderer } from '@ipeople/sahara-studio';
import { settings } from '../view-runtime/config';

const BLOCK_NAME = 'baseInfo';

export default {
  mixins: [Renderer],

  props: ['addition'],

  data() {
    return {
      settings,
    };
  },

  async created() {
    const {
      pageType, functionCode, pkId, regulations, viewNode, sourceType,
    } = this.addition || {};

    await this.launch({
      viewNode,
      regulations,
    });

    const { data: model = {} } = ['detail', 'update'].indexOf(pageType) > -1
      ? await req('getListEngineFormData', {
        pkId,
        functionCode,
      })
      : {};
    sourceType === 'md' && this.$forms.setModel({ [BLOCK_NAME]: model });
    sourceType === 'dm' && this.$forms.setModel(model);
  },

  rendered() {
    const { pageType } = this.addition || {};
    if (pageType === 'detail') {
      this.$forms.setReadonly(true);
    }
  },

  methods: {
    writeback(result) {
      return this.submit().then(result).catch(result);
    },
  },
};
</script>
