# 说明

配置平台总体的工作流程可以拆分为设计时和运行时两部分，在引入配置平台之后项目上可以通过直接配置配置平台提供的菜单路径访问设计器进行配置页面，在此基础上产品/项目团队需独立提供一套运行时文件用来承载设计数据并进行展示。

本章节主要讲解配置平台设计时部分

# 注意

虽然设计时的代码已内置在CPF包内，但存在需要注册进设计器的配置依赖需要在项目上实现

# 需要引入设计器的项目定制组件

在main.js中添加以下代码

import Engine from '@ipeople/sahara-studio';
import EngineDetails from '@/details'; //  定制业务组件的配置信息文件
import { initFieldHandlers } from '@/fssc/cpf/views/view-design/config';

Vue.use(Engine, {
  initFieldHandlers,
  details: EngineDetails,
  componentList: [
    // 将需要注册的表单业务组件配置在这里
    { tag: 'm-city-picker', label: '地区选择器' },
  ],
  tableEditorList: [
    // 将需要注册的表格业务组件配置在这里
    { tag: 'm-city-picker', label: '地区选择器', type: 'custom' },
  ],
});

# 需要引入设计器的配置信息清单

设计时（designSettings）

import req from '@/api/cpf'

const uiTypeMap = {
  '02': 'el-input',
  '03': 'el-input-number',
  '04': 'el-date-picker'
}

export const getUIType = node => node.uiType ||
  (node.dictCode && 'i-selector') ||
  uiTypeMap[node.dataType] ||
  'el-input'

const fetchTemplateRenderers = (templateType, functionName) => {
  return req('postFunctionsPage', {
    pageIndex: 1,
    pageSize: 9999,
    isPublish: 'Y',
    templateType,
    functionName
  }, true, false).then(res => {
    console.log(111, res)
    const data = res.data
    const pfx = templateType !== 'FORM' ? '' : `${templateType[0]}.`
    return data.filter(el => el.isPublish === 'Y').map(el => ({
      label: el.functionName,
      value: `${pfx}${el.functionCode}`,
      functionId: el.functionId
    }))
  })
}

const fetchOrderRenderers = fetchTemplateRenderers.bind(null, 'ORDER')

const fetchFormRenderers = fetchTemplateRenderers.bind(null, 'FORM')

const fetchAysncPagePaths = () => {
  return Promise.all([fetchOrderRenderers(), fetchFormRenderers()])
    .then(([orderData, formData]) => {
      return [].concat(orderData, formData)
    })
}
// 台账选择器配置
export const fetchListRendererConfig = async (param) => {
  const { data } = await req('getListTemplates', { functionId: param.functionId })
  const pkColumn = data.pkColumn.split('.').splice(-1)[0] // 节选末位字段作为主键名

  return {
    title: data.functionName,
    valueKey: pkColumn,
    props: {
      value: pkColumn,
      label: pkColumn
    },
    // 出参字段：通过回写抛出的数据
    resultKeys: (data.tableNodeList || []).map(el => ({
      label: el.nodeName,
      value: el.nodeKey.split('.').slice(-1)[0] // 表格取末位字段
    })),
    // 入参字段：通过附加信息传入的数据
    paramKeys: (data.searchNodeList || []).map(el => ({
      label: el.nodeName,
      value: el.nodeKey.split('.').join('$') // 表单替换拼接符
    }))
  }
}

const pageTypes = [
  { label: '填单页', value: 'edit' }
]

// 这里写自定义渲染函数的配置属性，用于自定义工具栏渲染
const customRenderer = [
  {
    label: '计算列表合计值', // label
    value: 'computeTableColumnTotal', // 函数名，需要在相对应的运行时实现该函数体
    options: [ // 函数支持的参数选项
      { label: 'xxx合计', value: 'text' },
      { label: '列表字段编码', value: 'columnProp' }
    ]
  }
]

const _updateViewAsync = async (type, code, cb) => {
  if (type === 'renderer.list') {
    // 在台账中，不存在构建所需要的 functionId 信息
    const { data } = await req('postFunctionsPage', { functionCode: code, pageIndex: 1, pageSize: 1 })
    const [param] = data.data || []
    const { valueKey, props } = await fetchListRendererConfig(param)
    cb(Promise.resolve({ valueKey, props }))
  }
  if (type === 'renderer.dimension') {
    // const { valueKey, props } = await fetchRendererConfig(type)({ value: code });
    // cb({ valueKey, props });

    cb(Promise.resolve({
      valueKey: 'entityId',
      props: {
        value: 'entityId',
        label: 'entityBaseAttributeName'
      }
    }))
  }
}

export const initFieldHandlers = {
  'i-lookup': res => {
    const [controller] = res.children
    const { businessType, businessCode: rendererCode } = controller.meta

    if (!controller.props.business && businessType) {
      const business = `renderer.${businessType.toLowerCase()}`
      controller.props.business = business
      controller.attrs.rendererCode = rendererCode

      _updateViewAsync(business, rendererCode, (data) => {
        Object.assign(controller.props, data)
      })
    }
  },
  'i-selector': res => {
    const [controller] = res.children
    const { businessType, businessCode: rendererCode } = controller.meta

    if (!controller.props.remoteKey && businessType) {
      const remoteKey = `renderer.${businessType.toLowerCase()}`
      controller.props.dataKey = 'remote'
      controller.props.remoteKey = remoteKey
      controller.attrs.rendererCode = rendererCode
      if (['renderer.list', 'renderer.dimension'].includes(remoteKey)) {
        controller.props.lazy = true // 默认使用懒加载
      }

      _updateViewAsync(remoteKey, rendererCode, (data) => {
        Object.assign(controller.props, data)
      })
    }
  }
}

export default {
  'route.pagePaths.async': fetchAysncPagePaths, //按钮功能为路由跳转时路由路径候选项
  'route.pageTypes': pageTypes, //按钮功能为路由跳转时页面类型候选项
  'sahara.common.custom.renderer': customRenderer // 自定义渲染函数，需要在相对应的运行时实现该函数体
}
